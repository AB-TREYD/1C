# Процедуры

## Процедура Печатать(Команда) 
```
	
	ТабДок = Новый ТабличныйДокумент;
	
	ПечататьНаСервере(ТабДок);
	
	ТабДок.Показать(); 
```

## Процедура ПечататьНаСервере(ТабДок) 
```

	Макет = Документы.СменноеЗаданиеУпаковкиШин.ПолучитьМакет("СменноеЗаданиеКоробок");

	ТабДок.ОриентацияСтраницы = ОриентацияСтраницы.Ландшафт;
	
	ОбластьЗаголовок = Макет.ПолучитьОбласть("Заголовок");  
	ТабДок.Вывести(ОбластьЗаголовок);
	
	ОбластьПодходящиеКоробкиШапка= Макет.ПолучитьОбласть("ПодходящиеКоробкиШапка");  
	ТабДок.Вывести(ОбластьПодходящиеКоробкиШапка);
	
	Для каждого Коробка из ЭтаФорма.Объект.КоробкиСменногоЗадания Цикл
		ОбластьПодходящиеКоробки = Макет.ПолучитьОбласть("ПодходящиеКоробки"); 
		
		ОбластьПодходящиеКоробки.Параметры.ВсеШиныВКоробке = Коробка.ВсеШиныВКоробке;
		ОбластьПодходящиеКоробки.Параметры.КоробкиЦПМ      = Коробка.КоробкиЦПМ;
		ОбластьПодходящиеКоробки.Параметры.ВесКоробки      = Коробка.ВесКоробки;
		ОбластьПодходящиеКоробки.Параметры.НомерСчета      = Коробка.НомерСчета;
		ОбластьПодходящиеКоробки.Параметры.Комментарий     = Коробка.Комментарий;
		//Ещё комент добавить
		
		ТабДок.Вывести(ОбластьПодходящиеКоробки);
	КонецЦикла;	
```

## Процедура ЗаполнитьФактУпаковки(Команда)

```
	
	ЗаказыДляВыбораКоробок = Новый Структура; 
	ЗаказыДляВыбораКоробок.Вставить("СсылкаНаКоробку",              Элементы.КоробкиСменногоЗадания.ТекущиеДанные.КоробкиЦПМ);
	ЗаказыДляВыбораКоробок.Вставить("СсылкаНаСгруппированныйЗаказ", Элементы.КоробкиСменногоЗадания.ТекущиеДанные.СсылкаНаСгруппированныйЗаказ);
	ЗаказыДляВыбораКоробок.Вставить("НомерСчета",                   Элементы.КоробкиСменногоЗадания.ТекущиеДанные.НомерСчета);
	ЗаказыДляВыбораКоробок.Вставить("СсылкаНаСменноеЗадание",       Объект.Ссылка);
	
	ОткрытьФорму("Документ.СменноеЗаданиеУпаковкиШин.Форма.ФормаВнесенияФакта", ЗаказыДляВыбораКоробок, ЭтаФорма,,,,,РежимОткрытияОкнаФормы.БлокироватьВесьИнтерфейс);
```
